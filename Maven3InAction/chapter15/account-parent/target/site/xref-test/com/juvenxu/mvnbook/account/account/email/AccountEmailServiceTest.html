18      private GreenMail greenMail;
19  
20      @Before
21      public void startMailServer() throws Exception {
22          greenMail = new GreenMail(ServerSetupTest.SMTP);
23          greenMail.setUser("hotenglish@163.com", "sin90=1");
24          greenMail.start();
25      }
26  
27      @Test
28      public void testSendMail() throws Exception {
29          ApplicationContext ctx = new ClassPathXmlApplicationContext("account-email.xml");
30          AccountEmailService accountEmailService = (AccountEmailService) ctx.getBean("accountEmailService");
31          String subject = "Test Subject";
32          String htmlText = "<h3>Test</h3>";
33          accountEmailService.sendMail("hotenglish@163.com", subject, htmlText);
34          greenMail.waitForIncomingEmail(2000, 1);
35          Message[] msg = greenMail.getReceivedMessages();
36          assertEquals(1, msg.length);
37          assertEquals(subject, msg[0].getSubject());
38          assertEquals(htmlText, GreenMailUtil.getBody(msg[0]).trim());
39      }
40  
41      @After
42      public void stopMailServer() throws Exception {
43          greenMail.stop();
44      }
