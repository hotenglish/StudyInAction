20      @Before
21      public void prepare() throws Exception {

26      @Test
27      public void testGenerateCaptcha() throws Exception {
28          String captchaKey = service.generateCaptchaKey();
29          assertNotNull(captchaKey);
30  
31          byte[] captchaImage = service.generateCaptchaImage(captchaKey);
32          assertTrue(captchaImage.length > 0);
33  
34          File image = new File("target/" + captchaKey + ".jpg");
35          OutputStream output = null;
36          try {
37              output = new FileOutputStream(image);
38              output.write(captchaImage);
39          } finally {
40              if (output != null) {
41                  output.close();
42              }
43          }
44          assertTrue(image.exists() && image.length() > 0);
45      }
46  
47      @Test
48      public void testValidateCaptchaCorrect() throws Exception {
49          List<String> preDefineTexts=new ArrayList<>();
50          preDefineTexts.add("12345");
51          preDefineTexts.add("abcde");
52          service.setPreDefinedTexts(preDefineTexts);
53  
54          String captchaKey=service.generateCaptchaKey();
55          service.generateCaptchaImage(captchaKey);
56          assertTrue(service.validateCaptcha(captchaKey,"12345"));
57  
58          captchaKey=service.generateCaptchaKey();
59          service.generateCaptchaImage(captchaKey);
60          assertTrue(service.validateCaptcha(captchaKey,"abcde"));
61      }
62  
63      @Test
64      public void testValidateCaptchaIncorrect() throws Exception {
