44      @Override
45      public void signUp(SignUpRequest signUpRequest) throws AccountServiceException {
46          try {
47              if (!signUpRequest.getPassword().equals(signUpRequest.getConfirmPassword())) {
48                  throw new AccountServiceException("2 passwords do not match.");
49              }
50  
51              if (!accountCaptchaService.validateCaptcha(signUpRequest.getCaptchaKey(),
52                      signUpRequest.getCaptchaValue())) {
53                  throw new AccountServiceException("Incorrect Captcha");
54              }
55  
56              Account account = new Account();
57              account.setId(signUpRequest.getId());
58              account.setEmail(signUpRequest.getEmail());
59              account.setName(signUpRequest.getName());
60              account.setPassword(signUpRequest.getPassword());
61              account.setActivated(false);
62  
63              accountPersistService.createAccount(account);
64  
65              String activationId = RandomGenerator.getRandomString();
66  
67              activationMap.put(activationId, account.getId());
68  
69              String link = signUpRequest.getActivateServiceUrl().endsWith("/") ? signUpRequest.getActivateServiceUrl()
70                      + activationId : signUpRequest.getActivateServiceUrl() + "?key=" + activationId;
71  
72              accountEmailService.sendMail(account.getEmail(),"Please Activate Your Account", link);
73          } catch (AccountCaptchaException e) {
74              throw new AccountServiceException("Unable to validate captcha.", e);
75          } catch (AccountPersistException e) {
76              throw new AccountServiceException("Unable to create account.", e);
77          } catch (AccountEmailException e) {
78              throw new AccountServiceException("Unable to send activation mail.", e);
79          }
80      }
81  
82      @Override
83      public void activate(String activationNumber) throws AccountServiceException {
84          String accountId = activationMap.get(activationNumber);
85  
86          if (accountId == null) {
87              throw new AccountServiceException("Invalid account activation ID.");
88          }
89  
90          try {
91              Account account = accountPersistService.readAccount(accountId);
92              account.setActivated(true);
93              accountPersistService.updateAccount(account);
94          } catch (AccountPersistException e) {
95              throw new AccountServiceException("Unable to activate account.");
96          }
97      }
98  
99      @Override
100     public void login(String id, String password) throws AccountServiceException {
101         try {
102             Account account = accountPersistService.readAccount(id);
103             if (account == null) {
104                 throw new AccountServiceException("Account does not exist.");
105             }
106 
107             if (!account.isActivated()) {
108                 throw new AccountServiceException("Account is disabled.");
109             }
110 
111             if (!account.getPassword().equals(password)) {
112                 throw new AccountServiceException("Incorrect password.");
113             }
114 
115         } catch (AccountPersistException e) {
116             throw new AccountServiceException("Unable to log in", e);
117         }
118     }
