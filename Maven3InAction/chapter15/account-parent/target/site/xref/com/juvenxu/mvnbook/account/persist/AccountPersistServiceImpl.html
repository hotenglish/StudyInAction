15  public class AccountPersistServiceImpl implements AccountPersistService {

17      private String file;

19      private static final String ELEMENT_ROOT = "account-persist";
20  
21      private static final String ELEMENT_ACCOUNTS = "accounts";
22  
23      private static final String ELEMENT_ACCOUNT = "account";
24  
25      private static final String ELEMENT_ACCOUNT_ID = "id";
26  
27      private static final String ELEMENT_ACCOUNT_NAME = "name";
28  
29      private static final String ELEMENT_ACCOUNT_EMAIL = "email";
30  
31      private static final String ELEMENT_ACCOUNT_PASSWORD = "password";
32  
33      private static final String ELEMENT_ACCOUNT_ACTIVATED = "activated";
34  
35      private SAXReader reader = new SAXReader();
36  
37      @Override
38      public Account createAccount(Account account) throws AccountPersistException {
39          int id = 0;
40          Document doc = readDocument();
41          Element accountsEle = doc.getRootElement().element(ELEMENT_ACCOUNTS);
42          for (Element item : (List<Element>) accountsEle.elements()) {
43              id = Integer.parseInt(item.elementText(ELEMENT_ACCOUNT_ID));
44          }
45          id++;
46          Element element = new UserDataElement(ELEMENT_ACCOUNT);
47          element.addElement(ELEMENT_ACCOUNT_ID).setText(String.valueOf(id));
48          element.addElement(ELEMENT_ACCOUNT_NAME).setText(account.getName());
49          element.addElement(ELEMENT_ACCOUNT_EMAIL).setText(account.getEmail());
50          element.addElement(ELEMENT_ACCOUNT_PASSWORD).setText(account.getPassword());
51          element.addElement(ELEMENT_ACCOUNT_ACTIVATED).setText(Boolean.toString(account.isActivated()));
52          accountsEle.add(element);
53          writeDocument(doc);
54          account.setId(String.valueOf(id));
55          return account;
56      }
57  
58      @Override
59      public Account readAccount(String id) throws AccountPersistException {
60          Document doc = readDocument();
61          Element accountsEle = doc.getRootElement().element(ELEMENT_ACCOUNTS);
62          for (Element accountEle : (List<Element>) accountsEle.elements()) {
63              if (accountEle.elementText(ELEMENT_ACCOUNT_ID).equals(id)) {
64                  return buildAccount(accountEle);
65              }
66          }
67          return null;
68      }
69  
70      @Override
71      public Account updateAccount(Account account) throws AccountPersistException {
72          Document doc = readDocument();
73          Element accountsEle = doc.getRootElement().element(ELEMENT_ACCOUNTS);
74          for (Element accountEle : (List<Element>) accountsEle.elements()) {
75              if (accountEle.elementText(ELEMENT_ACCOUNT_ID).equals(account.getId())) {
76                  accountEle.element(ELEMENT_ACCOUNT_NAME).setText(account.getName());
77                  accountEle.element(ELEMENT_ACCOUNT_EMAIL).setText(account.getEmail());
78                  accountEle.element(ELEMENT_ACCOUNT_PASSWORD).setText(account.getPassword());
79                  accountEle.element(ELEMENT_ACCOUNT_ACTIVATED).setText(Boolean.toString(account.isActivated()));
80              }
81          }
82          writeDocument(doc);
83          return account;
84      }
85  
86      @Override
87      public void deleteAccount(String id) throws AccountPersistException {
88          Document doc = readDocument();
89          Element accountsEle = doc.getRootElement().element(ELEMENT_ACCOUNTS);
90          for (Element accountEle : (List<Element>) accountsEle.elements()) {
91              if (accountEle.elementText(ELEMENT_ACCOUNT_ID).equals(id)) {
92                  accountsEle.remove(accountEle);
93                  writeDocument(doc);
94                  return;
95              }
96          }
97      }
98  
99      private Account buildAccount(Element element) {
100         Account account = new Account();
101         account.setId(element.elementText(ELEMENT_ACCOUNT_ID));
102         account.setName(element.elementText(ELEMENT_ACCOUNT_NAME));
103         account.setEmail(element.elementText(ELEMENT_ACCOUNT_EMAIL));
104         account.setPassword(element.elementText(ELEMENT_ACCOUNT_PASSWORD));
105         account.setActivated("true".equals(element.elementText(ELEMENT_ACCOUNT_ACTIVATED)) ? true : false);
106         return account;
107     }
108 
109     private Document readDocument() throws AccountPersistException {
110         File dataFile = new File(file);
111         if (!dataFile.exists()) {
112             dataFile.getParentFile().mkdirs();
113             Document doc = DocumentFactory.getInstance().createDocument();
114             Element rootEle = doc.addElement(ELEMENT_ROOT);
115             rootEle.addElement(ELEMENT_ACCOUNTS);
116             writeDocument(doc);
117         }
118 
119         try {
120             return reader.read(new File(file));
121         } catch (DocumentException e) {
122             throw new AccountPersistException("Unable to read persist data xml", e);
123         }
124     }
125 
126     private void writeDocument(Document doc) throws AccountPersistException {
127         Writer out = null;
128         try {
129             out = new OutputStreamWriter(new FileOutputStream(file), "utf-8");
130             XMLWriter writer = new XMLWriter(out, OutputFormat.createPrettyPrint());
131             writer.write(doc);
132         } catch (IOException e) {
133             throw new AccountPersistException("Unable to write persist data xml", e);
134         } finally {
135             try {
136                 if (out != null) {
137                     out.close();
138                 }
139             } catch (IOException e) {
140                 throw new AccountPersistException("Unable to close persist data writer", e);
141             }
142         }
143     }
