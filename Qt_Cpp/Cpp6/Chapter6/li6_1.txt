#include <iostream>

using namespace std;

class Complex{
    float real;
    float imag;
public:
    Complex(float r=0,float i=0){ real =r; imag=i;}
    void Show(){cout<<real<<"+"<<imag<<"j"<<endl;}
    Complex operator+(Complex &c);
};

Complex Complex::operator+(Complex &c){
    float r,i;
    r=real+c.real;
    i=imag+c.imag;
    return Complex(r,i);
}

int main()
{
    Complex x(5,3);
    Complex y(4,2);
    Complex z;
    z=x+y;
    z.Show();
    return 0;
}
