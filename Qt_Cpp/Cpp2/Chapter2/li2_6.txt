//使用if语句嵌套形式
#include <iostream>

using namespace std;

int main(){
    int a,b,c,max;
    cout<<"\n please input three numbers:";
    cin>>a>>b>>c;

    if(a<b){
        if(b<c)
            max=c;
        else
            max=b;
    }else{
        if(a<c)
            max=c;
        else
            max=a;
    }
    cout<<"max="<<max<<endl;
    return 0;
}
