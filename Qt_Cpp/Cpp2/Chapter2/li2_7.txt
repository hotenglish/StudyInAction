//使用switch语句实现多分支结构
#include <iostream>

using namespace std;

int main(){
   int score;    //百分制成绩
   cout<<"请输入一个百分制成绩:";
   cin>>score;
   if(score<0||score>100){
       cout<<"input error\n";
       exit(1);
   }
   switch(score/10){
      case 10:
      case 9:cout<<"A\n";break;
      case 8:cout<<"B\n";break;
      case 7:cout<<"C\n";break;
      case 6:cout<<"D\n";break;
      default:cout<<"E\n";
   }
   return 0;
}
