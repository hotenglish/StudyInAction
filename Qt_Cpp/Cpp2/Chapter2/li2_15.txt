#include <iostream>
#include <string>
using namespace std;
const int N=10;
//冒泡排序函数
void Bubblesort(string name[N],int n)
{
    string str;
    for(int j=1;j<=n-1;j++){
        for(int i=0;i<n-j-1;i++){
            if(name[i]>name[i+1]){
                str=name[i];
                name[i]=name[i+1];
                name[i+1]=str;
            }
        }
    }
}

int main(){
    string name[N];
    int i;
    cout<<"请任意输入5个国家名字\n";
    for(i=0;i<5;i++){
        cin>>name[i];
    }
    Bubblesort(name,5);
    cout<<"按字线排序后的名字顺序:\n";
    for(i=0;i<5;i++)
        cout<<name[i]<<endl;
    return 0;

}
