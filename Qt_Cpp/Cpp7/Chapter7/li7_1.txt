//li7_1.cpp
#include <iostream>
using namespace std;
template<class T>             //函数模板声明
void Swap(T &a,T &b);

int main()
{
    int i1=3,i2=5;
    Swap(i1,i2);
    cout << "i1="<<i1<<",i2="<<i2<< endl;
    double d1=2.1,d2=5.3;
    Swap(d1,d2);
    cout << "d1="<<d1<<",d2="<<d2<< endl;
    char c1='m',c2='n';
    Swap(c1,c2);
    cout << "c1="<<'\''<<c1<<'\''<<",c2="<<'\''<<c2<<'\''<< endl;
    return 0;
}

template<class T>             //函数模板定义
void Swap(T &a,T &b){
    T temp;
    temp=a;
    a=b;
    b=temp;
}
