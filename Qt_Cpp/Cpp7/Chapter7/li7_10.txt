//li7_10.cpp
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main(){
    int num=2;
    vector<int> v1;
    for(int i=0;i<5;i++){
        v1.push_back(2*i);         //填充序列0 2 4 6 8
    }
    vector<int>::iterator iter;    //定义迭代器iter
    iter=find(v1.begin(),v1.end(),num);
    if(iter==v1.end()){
        cout<<"未找到元素"<<num<<endl;
    }else{
        cout<<"匹配元素的索引值是"<<iter-v1.begin()<<endl;
    }
    return 0;
}
