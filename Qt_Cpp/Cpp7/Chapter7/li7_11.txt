#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main(){
    vector<int> v;
    v.push_back(8);
    v.push_back(3);
    v.push_back(-16);
    v.push_back(70);
    v.push_back(90);
    vector<int>::iterator iter;
    for(iter=v.begin();iter!=v.end();iter++)
        cout<<*iter<<' ';
    cout<<endl;
    sort(v.begin(),v.end());
    cout<<"排序后：?\n";
    for(iter=v.begin();iter!=v.end();iter++)
        cout<<*iter<<' ';
    cout<<endl;
    return 0;
}
