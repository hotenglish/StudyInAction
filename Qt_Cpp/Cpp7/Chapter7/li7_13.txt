#include <iostream>
using namespace std;
double Div(double x,double y){
    if(y==0)
        throw y;                                    //抛出异常
    return x/y;
}

int main(){
    try{
        //被检查的语句
        cout<<Div(6,4)<<endl;
        cout<<Div(5,0)<<endl;
        cout<<Div(9,4)<<endl;
    }catch(double){                                //捕获异常
        cout<<"except of deviding zero."<<endl;    //异常处理语句
    }
    return 0;
}
