//li3_22.cpp
#include <iostream>
using namespace std;
class Student{
public:
    Student();    //构造函数
    void Init();  //成员函数声明
    void Disp();  //成员函数声明
    static int number;  //静态数据成员定义
private:
    int num;
    char name[10];
    int age;
};

int Student::number=0;   //静态数据成员初始化

Student::Student(){
    number+=1;
}

void Student::Init(){
    cout<<"请输入学号：";
    cin>>num;
    cout<<"请输入姓名：";
    cin>>name;
    cout<<"请输入年龄：";
    cin>>age;
}

void Student::Disp(){
    cout<<"学生的基本信息："<<endl;
    cout<<"学号："<<num;
    cout<<"姓名："<<name;
    cout<<"年龄："<<age;
    cout<<endl;
}

int main(){
    cout<<"学生人数："<<endl;
    cout<<Student::number<<endl;    //通过类名访问静态成员
    Student stu1;
    cout<<Student::number<<endl;    //通过类名访问静态成员
    cout<<stu1.number<<endl;        //通过对象名访问静态成员
    Student stu2;
    cout<<Student::number<<endl;    //通过类名访问静态成员
    cout<<stu1.number<<endl;        //通过对象名访问静态成员
    cout<<stu2.number<<endl;        //通过对象名访问静态成员
}
