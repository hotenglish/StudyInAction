#include <iostream>

using namespace std;

class Student{
    // 数据成员
    int num;
    char name[10];
    int age;
    // 成员函数
    void init(){
        cout<<"请输入学号：";
        cin>>num;
        cout<<"请输入姓名：";
        cin>>name;
        cout<<"请输入年龄：";
        cin>>age;
    }
    // 成员函数
    void Disp(){
        cout<<"学生的基本信息："<<endl;
        cout<<"学号："<<num;
        cout<<"姓名："<<name;
        cout<<"年龄："<<age;
        cout<<endl;
    }

};

int main()
{
    Student stu1,stu2;    //用类Student定义对象stu1和stu2
    stu1.init();          //对象调用成员函数，但编译出错
    stu1.Disp();          //编译出错
    Student *pstu=&stu2;  //定义指向对象的指针
    pstu->init();         //编译出错
    pstu->Disp();         //编译出错
    return 0;
}
