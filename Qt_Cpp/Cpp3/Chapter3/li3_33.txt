//li3_33.cpp
#include <iostream>
using namespace std;
#include <cstring>
class Student{
    public:
        Student(int nu,char *na,int ag);        //带参构造函数
        ~Student();
        void Init();                            //成员函数声明
        void Disp();                            //成员函数声明
    private:
        int num;
        char name[10];
        int age;
};

Student::Student(int nu,char *na,int ag){
    num=nu;
    strcpy(name,na);
    age=ag;
}

Student::~Student(){
    cout<<"Destructor called"<<endl;
}

void Student::Init(){
    cout<<"请输入学号：";                              //成员函数定义
    cin>>num;
    cout<<"请输入姓名：";
    cin>>name;
    cout<<"请输入年龄：";
    cin>>age;
}

void Student::Disp(){                               //成员函数定义
    cout<<"学生的基本信息："<<endl;
    cout<<"学号："<<num;
    cout<<"姓名："<<name;
    cout<<"年龄："<<age;
    cout<<endl;
}

int main(){
    Student stu[3]={Student(201201,"李平",19),Student(201202,"王丽",19),Student(201203,"宋强",20)};
    for(int i=0;i<3;i++){
        cout<<"第"<<i+1<<"个";
        stu[i].Disp();                              //对象元素调用成员函数
    }
    return 0;
}
