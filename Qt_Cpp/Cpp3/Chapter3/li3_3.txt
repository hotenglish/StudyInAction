#include <iostream>
using namespace std;

class Student{
    // 数据成员
    int num;
    char name[10];
    int age;
public:               //公有访问权限
    void init(){      //成员函数
        cout<<"请输入学号：";
        cin>>num;
        cout<<"请输入姓名：";
        cin>>name;
        cout<<"请输入年龄：";
        cin>>age;
    }
    void Disp(){       //成员函数
        cout<<"学生的基本信息："<<endl;
        cout<<"学号："<<num;
        cout<<"姓名："<<name;
        cout<<"年龄："<<age;
        cout<<endl;
    }

};

int main()
{
    Student stu1,stu2;        //用类Student定义对象stu1和stu2
    stu1.init();              //通过对象名调用成员函数
    stu1.Disp();              //通过对象名调用成员函数
    Student *pstu=&stu2;      //定义指向对象的指针
    pstu->init();             //通过对象指针调用成员函数
    pstu->Disp();             //通过对象指针调用成员函数
    return 0;
}
